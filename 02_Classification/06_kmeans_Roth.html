import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

from sklearn.datasets import make_blobs
from sklearn.cluster import KMeans
from sklearn.metrics import silhouette_samples, silhouette_score

X = make_blobs(n_samples = 400, n_features = 2, centers = 5, cluster_std = 1, random_state=42)
X = X[0]
plt.scatter(X[:,0],X[:,1])
plt.show()

cluster_numbers = [2, 3, 4, 5, 6, 7, 8]
scores = []

fig, ax = plt.subplots(len(cluster_numbers), 1, figsize = (4, 20))
for k,cn in enumerate(cluster_numbers):
    km = KMeans(n_clusters = cn)
    km.fit(X)

    labels = km.labels_
    centroids = km.cluster_centers_
    score = silhouette_score(X, labels)
    scores.append(score)
    
    ax[k].scatter(X[:,0], X[:,1], c = labels, s = 2)
    ax[k].plot(centroids[:,0], centroids[:,1], 'rx')
    ax[k].text(0,-7, f'{score:0.3f}')

plt.show()

sil_score = silhouette_score(X, labels)
sil_samples = silhouette_samples(X,labels)
